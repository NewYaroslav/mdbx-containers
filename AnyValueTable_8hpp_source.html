253 template<typename F>
254 void with_transaction(F&& action, TransactionMode mode, MDBX_txn* txn) const {
255 if (txn) {
256 action(txn);
257 return;
258 }
259 txn = thread_txn();
260 if (txn) {
261 action(txn);
262 return;
263 }
264 auto txn_guard = m_connection->transaction(mode);
265 try {
266 action(txn_guard.handle());
267 txn_guard.commit();
268 } catch (...) {
269 try { txn_guard.rollback(); } catch (...) {}
270 throw;
271 }

275 bool put_typed(const KeyT& key, const T& value, bool upsert, MDBX_txn* txn) {
276 SerializeScratch sc_key;
277 SerializeScratch sc_value;
278 MDBX_val db_key = serialize_key(key, sc_key);
279 MDBX_val raw_val = serialize_value(value, sc_value);
280 MDBX_val db_val = wrap_with_type_tag<T>(raw_val);
281 MDBX_put_flags_t flags = upsert ? MDBX_UPSERT : MDBX_NOOVERWRITE;
282 int rc = mdbx_put(txn, m_dbi, &db_key, &db_val, flags);
283 if (!upsert && rc == MDBX_KEYEXIST) {
284 return false;
285 }
286 check_mdbx(rc, upsert ? "Failed to set value" : "Failed to insert value");
287 return true;
288 }

290 template <class T>
291 bool get_typed(const KeyT& key, T& out, MDBX_txn* txn) const {
292 SerializeScratch sc_key;
293 MDBX_val db_key = serialize_key(key, sc_key);
294 MDBX_val db_val{};
295 int rc = mdbx_get(txn, m_dbi, &db_key, &db_val);
296 if (rc == MDBX_NOTFOUND) return false;
297 check_mdbx(rc, "Failed to retrieve value");
298 MDBX_val checked = unwrap_and_check_type_tag<T>(db_val);
299 out = deserialize_value<T>(checked);
300 return true;

323 void db_list_keys(std::vector<KeyT>& out, MDBX_txn* txn) const {
324 MDBX_cursor* cursor = nullptr;
325 check_mdbx(mdbx_cursor_open(txn, m_dbi, &cursor), "Failed to open MDBX cursor");
326 MDBX_val db_key, db_val;
327 while (mdbx_cursor_get(cursor, &db_key, &db_val, MDBX_NEXT) == MDBX_SUCCESS) {
328 out.emplace_back(deserialize_value<KeyT>(db_key));
329 }
330 mdbx_cursor_close(cursor);
