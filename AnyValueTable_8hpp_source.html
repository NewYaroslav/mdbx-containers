217 template<typename F>
218 void with_transaction(F&& action, TransactionMode mode, MDBX_txn* txn) const {
219 if (txn) {
220 action(txn);
221 return;
222 }
223 txn = thread_txn();
224 if (txn) {
225 action(txn);
226 return;
227 }
228 auto txn_guard = m_connection->transaction(mode);
229 try {
230 action(txn_guard.handle());
231 txn_guard.commit();
232 } catch (...) {
233 try { txn_guard.rollback(); } catch (...) {}
234 throw;
235 }
236 }

238 template <class T>
239 bool put_typed(const KeyT& key, const T& value, bool upsert, MDBX_txn* txn) {
240 MDBX_val db_key = serialize_key(key);
241 MDBX_val raw_val = serialize_value(value);
242 MDBX_val db_val = wrap_with_type_tag<T>(raw_val);
243 MDBX_put_flags_t flags = upsert ? MDBX_UPSERT : MDBX_NOOVERWRITE;
244 int rc = mdbx_put(txn, m_dbi, &db_key, &db_val, flags);
245 if (!upsert && rc == MDBX_KEYEXIST) {
246 return false;
247 }
248 check_mdbx(rc, upsert ? "Failed to set value" : "Failed to insert value");
249 return true;
250 }

252 template <class T>
253 bool get_typed(const KeyT& key, T& out, MDBX_txn* txn) const {
254 MDBX_val db_key = serialize_key(key);
255 MDBX_val db_val{};
256 int rc = mdbx_get(txn, m_dbi, &db_key, &db_val);
257 if (rc == MDBX_NOTFOUND) return false;
258 check_mdbx(rc, "Failed to retrieve value");
259 MDBX_val checked = unwrap_and_check_type_tag<T>(db_val);
260 out = deserialize_value<T>(checked);
261 return true;
262 }

282 void db_list_keys(std::vector<KeyT>& out, MDBX_txn* txn) const {
283 MDBX_cursor* cursor = nullptr;
284 check_mdbx(mdbx_cursor_open(txn, m_dbi, &cursor), "Failed to open MDBX cursor");
285 MDBX_val db_key, db_val;
286 while (mdbx_cursor_get(cursor, &db_key, &db_val, MDBX_NEXT) == MDBX_SUCCESS) {
287 out.emplace_back(deserialize_value<KeyT>(db_key));
288 }
289 mdbx_cursor_close(cursor);
