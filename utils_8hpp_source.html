176 [[nodiscard]] inline MDBX_val view_copy(const void* p, size_t n) {
177 bytes.clear();
178 bytes.resize(n);
179 if (n) std::memcpy(bytes.data(), p, n);
180 MDBX_val v;

184 }

188 MDBX_val v;

190 v.iov_len = bytes.size();
191 return v;
192 }

345 template <size_t N>
346 inline MDBX_val serialize_key(const std::bitset<N>& data, SerializeScratch& sc) {
347 const size_t num_bytes = (N + 7) / 8;
348 std::array<uint8_t, (N + 7) / 8> buffer;
349 buffer.fill(0);
350 for (size_t i = 0; i < N; ++i) {
351 if (data[i]) buffer[i / 8] |= (1 << (i % 8));
352 }
353 return sc.view_copy(buffer.data(), num_bytes);
354 }

596 typename std::enable_if<
597 has_value_type<T>::value &&
598 std::is_same<typename T::value_type, std::string>::value,
599 T>::type
600 deserialize_value(const MDBX_val& val) {
601 const uint8_t* ptr = static_cast<const uint8_t*>(val.iov_base);
602 const uint8_t* end = ptr + val.iov_len;
603
604 T result;
605 while (ptr + sizeof(uint32_t) <= end) {
606 uint32_t len;
607 std::memcpy(&len, ptr, sizeof(uint32_t));
608 ptr += sizeof(uint32_t);
609
610 if (ptr + len > end)
611 throw std::runtime_error("deserialize_value: corrupted data (length overflow)");
612
613 result.emplace_back(reinterpret_cast<const char*>(ptr), len);
614 ptr += len;
615 }
616
617 if (ptr != end)
618 throw std::runtime_error("deserialize_value: trailing data after deserialization");
619
620 return result;
621 }
