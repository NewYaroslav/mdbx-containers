171 inline std::string utf8_to_ansi(const std::string& utf8) noexcept {
172#ifdef _WIN32
173 int n_len = MultiByteToWideChar(CP_UTF8, 0, utf8.c_str(), -1, NULL, 0);
174 if (n_len == 0) return {};
175
176 std::wstring wide_string(n_len + 1, L'\0');
177 MultiByteToWideChar(CP_UTF8, 0, utf8.c_str(), -1, &wide_string[0], n_len);
178
179 n_len = WideCharToMultiByte(CP_ACP, 0, wide_string.c_str(), -1, NULL, 0, NULL, NULL);
180 if (n_len == 0) return {};
181
182 std::string ansi_string(n_len - 1, '\0');
183 WideCharToMultiByte(CP_ACP, 0, wide_string.c_str(), -1, &ansi_string[0], n_len, NULL, NULL);
184 return ansi_string;
185#else
186 return utf8;
187#endif
188 }

252# ifdef _WIN32
253 const char SEP = '\\';
254# else
255 const char SEP = '/';
256# endif

258 const size_t n = in.size();
259 size_t i = 0;
260
261 // ----- parse prefix (root) -----
262# ifdef _WIN32
263 bool is_unc = false;
264 std::string drive; // "C:"
265 std::vector<std::string> root_parts; // for UNC: server, share
266

269 is_unc = true;
270 i = 2;
271 // server
272 size_t start = i;
273 while (i < n && !is_path_sep(s[i])) ++i;

275 if (i < n && is_path_sep(s[i])) ++i;
276 // share
277 start = i;

279 if (i > start) root_parts.push_back(in.substr(start, i - start));
280 // пропускаем дополнительные слеши
281 while (i < n && is_path_sep(s[i])) ++i;

284 else if (n >= 2 && std::isalpha(static_cast<unsigned char>(s[0])) && s[1] == ':') {
285 drive.assign(in, 0, 2);

287 // пропустим один/несколько разделителей, если есть
288 while (i < n && is_path_sep(s[i])) ++i;
289 }

291 (!drive.size() && n >= 1 && is_path_sep(s[0]));
292# else
293 const bool is_abs_root = (n >= 1 && s[0] == '/');

309# ifdef _WIN32
310 && false
311# endif
312 ) { /* защищённая зона настроена ниже */ }
313# endif

315 // нельзя уходить выше корня абсолютного пути
316 comps.pop_back();
317 } else {
318 // относительный путь или уже нет куда сворачивать
319 comps.push_back("..");
320 }
321 } else {
322 comps.push_back(token);
323 }
324 };
325
326# ifdef _WIN32
327 // Для UNC защищаем первые два компонента (server/share) от свёртки ".."
328 size_t protected_count = 0;
329 if (is_unc) {
330 for (size_t k = 0; k < root_parts.size(); ++k) {
331 comps.push_back(root_parts[k]);
332 }
333 protected_count = comps.size();
334 }
335# endif
336
337 while (i < n) {
338 // пропускаем повторные разделители
339 while (i < n && is_path_sep(s[i])) ++i;
340 if (i >= n) break;
341 size_t start = i;
342 while (i < n && !is_path_sep(s[i])) ++i;
343 std::string token = in.substr(start, i - start);
344
345 if (token == "..") {
346# ifdef _WIN32
347 if (!comps.empty() && comps.size() > protected_count && comps.back() != "..") {
348 comps.pop_back();
349 } else if (!is_abs_root || (is_unc && comps.size() > protected_count)) {
350 comps.push_back("..");
351 }
352# else
353 if (is_abs_root) {
354 if (!comps.empty() && comps.back() != "..") comps.pop_back();
355 } else {
356 comps.push_back("..");
357 }
358# endif
359 } else if (token != ".") {
360 comps.push_back(token);
361 }
362 }

365 std::string out;
366
367# ifdef _WIN32
368 if (is_unc) {
369 out = "\\\\";

374 return out;

380 if (drive_absolute) out += SEP;
381 } else if (is_abs_root) {
382 out.push_back(SEP);
383 }
384# else

386# endif

394# ifdef _WIN32

396# else

413 inline PathComponents split_path(const std::string& path) {
414 PathComponents result;
415 size_t i = 0;
416 size_t n = path.size();
417
418 // Handle root paths for Unix and Windows
419 if (n >= 1 && (path[0] == '/' || path[0] == '\\')) {
420 // Unix root "/"
421 result.root = "/";
422 ++i;
423 } else if (n >= 2 && std::isalpha(path[0]) && path[1] == ':') {
424 // Windows drive letter "C:"
425 result.root = path.substr(0, 2);
426 i = 2;
427 if (n >= 3 && (path[2] == '/' || path[2] == '\\')) {
428 // "C:/"
429 ++i;
430 }
431 }
432
433 // Split the path into components
434 while (i < n) {
435 // Skip path separators
436 while (i < n && (path[i] == '/' || path[i] == '\\')) {
437 ++i;
438 }
439 // Find the next separator
440 size_t j = i;
441 while (j < n && path[j] != '/' && path[j] != '\\') {
442 ++j;
443 }
444 if (i < j) {
445 result.components.push_back(path.substr(i, j - i));
446 i = j;
447 }
448 }
449
450 return result;
451 }

456 inline void create_directories(const std::string& path) {
457 if (path.empty()) return;
458 PathComponents path_pc = split_path(get_parent_path(path));
459 auto &components = path_pc.components;
460 size_t components_size = components.size();
461
462 // Build the path incrementally and create directories
463 std::string current_path = path_pc.root;
464 for (size_t i = 0; i < components_size; ++i) {
465 if (!current_path.empty() && current_path.back() != '/' && current_path.back() != '\\') {
466 current_path += '/';
467 }
468 current_path += components[i];
469
470 // Skip special components
471 if (components[i] == ".." ||
472 components[i] == "/" ||
473 components[i] == "~/") continue;
474# ifdef _WIN32
475 int ret = _mkdir(utf8_to_ansi(current_path).c_str());
476# else
477 int ret = mkdir(current_path.c_str(), 0755);
478# endif
479 int errnum = errno;
480 if (ret != 0 && errnum != EEXIST) {
481 throw std::runtime_error("Failed to create directory: " + current_path);
482 }
483 }
484 }
