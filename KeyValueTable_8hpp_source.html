585 template<typename F>
586 void with_transaction(F&& action, TransactionMode mode = TransactionMode::WRITABLE, MDBX_txn* txn = nullptr) const {
587 if (txn) {
588 action(txn);
589 return;
590 }
591 txn = thread_txn(); // reuse transaction bound to this thread if any
592 if (txn) {
593 action(txn);
594 return;
595 }
596
597 auto txn_guard = m_connection->transaction(mode);
598 try {
599 action(txn_guard.handle());
600 txn_guard.commit();
601 } catch(...) {
602 // Ensure rollback on any exception and rethrow to the caller
603 try {
604 txn_guard.rollback();
605 } catch(...) {
606 // Ignore rollback errors here
607 }
608 throw;
609 }
610 }

618 void db_load(ContainerT<KeyT, ValueT>& container, MDBX_txn* txn_handle) {
619 MDBX_cursor* cursor = nullptr;
620 check_mdbx(mdbx_cursor_open(txn_handle, m_dbi, &cursor), "Failed to open MDBX cursor");
621
622 MDBX_val db_key, db_val;
623 while (mdbx_cursor_get(cursor, &db_key, &db_val, MDBX_NEXT) == MDBX_SUCCESS) {
624 auto&& key = deserialize_value<KeyT>(db_key);
625 auto&& value = deserialize_value<ValueT>(db_val);
626 container.emplace(std::move(key), std::move(value));
627 }
628
629 mdbx_cursor_close(cursor);
630 }

733 template<template <class...> class ContainerT>
734 void db_reconcile(const ContainerT<KeyT, ValueT>& container, MDBX_txn* txn_handle) {
735 // 1. Collect all keys from the container
736 std::unordered_set<KeyT> new_keys;
737 for (const auto& pair : container) {
738 new_keys.insert(pair.first);
739 MDBX_val db_key = serialize_key(pair.first);
740 MDBX_val db_val = serialize_value(pair.second);
741 check_mdbx(
742 mdbx_put(txn_handle, m_dbi, &db_key, &db_val, MDBX_UPSERT),
743 "Failed to write record"
744 );
745 }
746
747 // 2. Iterate over existing keys in the DB and remove the extras
748 MDBX_cursor* cursor = nullptr;
749 check_mdbx(mdbx_cursor_open(txn_handle, m_dbi, &cursor), "Failed to open MDBX cursor");
750
751 MDBX_val db_key, db_val;
752 while (mdbx_cursor_get(cursor, &db_key, &db_val, MDBX_NEXT) == MDBX_SUCCESS) {
753 auto&& key = deserialize_value<KeyT>(db_key);
754 if (new_keys.find(key) == new_keys.end()) {
755 check_mdbx(mdbx_cursor_del(cursor, MDBX_CURRENT), "Failed to delete record using cursor");
756 }
757 }
758
759 mdbx_cursor_close(cursor);
760 }

767 void db_reconcile(const std::vector<std::pair<KeyT, ValueT>>& container, MDBX_txn* txn_handle) {
768 // 1. Collect keys and upsert values
769 std::unordered_set<KeyT> new_keys;
770 for (size_t i = 0; i < container.size(); ++i) {
771 const KeyT& key = container[i].first;
772 const ValueT& value = container[i].second;
773
774 new_keys.insert(key);
775
776 MDBX_val db_key = serialize_key(key);
777 MDBX_val db_val = serialize_value(value);
778 check_mdbx(
779 mdbx_put(txn_handle, m_dbi, &db_key, &db_val, MDBX_UPSERT),
780 "Failed to write record"
781 );
782 }
783
784 // 2. Delete stale keys from DB
785 MDBX_cursor* cursor = nullptr;
786 check_mdbx(mdbx_cursor_open(txn_handle, m_dbi, &cursor), "Failed to open MDBX cursor");
787
788 MDBX_val db_key, db_val;
789 while (mdbx_cursor_get(cursor, &db_key, &db_val, MDBX_NEXT) == MDBX_SUCCESS) {
790 KeyT key = deserialize_value<KeyT>(db_key);
