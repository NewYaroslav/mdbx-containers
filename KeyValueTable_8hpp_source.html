577 template<typename F>
578 void with_transaction(F&& action, TransactionMode mode = TransactionMode::WRITABLE, MDBX_txn* txn = nullptr) const {
579 if (txn) {
580 action(txn);
581 return;
582 }
583 txn = thread_txn(); // reuse transaction bound to this thread if any
584 if (txn) {
585 action(txn);
586 return;
587 }
588
589 auto txn_guard = m_connection->transaction(mode);
590 try {
591 action(txn_guard.handle());
592 txn_guard.commit();
593 } catch(...) {
594 // Ensure rollback on any exception and rethrow to the caller
595 try {
596 txn_guard.rollback();
597 } catch(...) {
598 // Ignore rollback errors here
599 }
600 throw;
601 }

610 void db_load(ContainerT<KeyT, ValueT>& container, MDBX_txn* txn_handle) {
611 MDBX_cursor* cursor = nullptr;
612 check_mdbx(mdbx_cursor_open(txn_handle, m_dbi, &cursor), "Failed to open MDBX cursor");
613
614 MDBX_val db_key, db_val;
615 while (mdbx_cursor_get(cursor, &db_key, &db_val, MDBX_NEXT) == MDBX_SUCCESS) {
616 auto&& key = deserialize_value<KeyT>(db_key);
617 auto&& value = deserialize_value<ValueT>(db_val);
618 container.emplace(std::move(key), std::move(value));
619 }
620
621 mdbx_cursor_close(cursor);
622 }

724 template<template <class...> class ContainerT>
725 void db_reconcile(const ContainerT<KeyT, ValueT>& container, MDBX_txn* txn_handle) {
726 // 1. Collect all keys from the container
727 std::unordered_set<KeyT> new_keys;
728 for (const auto& pair : container) {
729 new_keys.insert(pair.first);
730 MDBX_val db_key = serialize_key(pair.first);
731 MDBX_val db_val = serialize_value(pair.second);
732 check_mdbx(
733 mdbx_put(txn_handle, m_dbi, &db_key, &db_val, MDBX_UPSERT),
734 "Failed to write record"
735 );
736 }
737
738 // 2. Iterate over existing keys in the DB and remove the extras
739 MDBX_cursor* cursor = nullptr;
740 check_mdbx(mdbx_cursor_open(txn_handle, m_dbi, &cursor), "Failed to open MDBX cursor");
741
742 MDBX_val db_key, db_val;
743 while (mdbx_cursor_get(cursor, &db_key, &db_val, MDBX_NEXT) == MDBX_SUCCESS) {
744 auto&& key = deserialize_value<KeyT>(db_key);
745 if (new_keys.find(key) == new_keys.end()) {
746 check_mdbx(mdbx_cursor_del(cursor, MDBX_CURRENT), "Failed to delete record using cursor");
747 }
748 }
749
750 mdbx_cursor_close(cursor);
751 }

758 void db_reconcile(const std::vector<std::pair<KeyT, ValueT>>& container, MDBX_txn* txn_handle) {
759 // 1. Collect keys and upsert values
760 std::unordered_set<KeyT> new_keys;
761 for (size_t i = 0; i < container.size(); ++i) {
762 const KeyT& key = container[i].first;
763 const ValueT& value = container[i].second;
764
765 new_keys.insert(key);
766
767 MDBX_val db_key = serialize_key(key);
768 MDBX_val db_val = serialize_value(value);
769 check_mdbx(
770 mdbx_put(txn_handle, m_dbi, &db_key, &db_val, MDBX_UPSERT),
771 "Failed to write record"
772 );
773 }
774
775 // 2. Delete stale keys from DB
776 MDBX_cursor* cursor = nullptr;
777 check_mdbx(mdbx_cursor_open(txn_handle, m_dbi, &cursor), "Failed to open MDBX cursor");
778
779 MDBX_val db_key, db_val;
780 while (mdbx_cursor_get(cursor, &db_key, &db_val, MDBX_NEXT) == MDBX_SUCCESS) {
781 KeyT key = deserialize_value<KeyT>(db_key);
